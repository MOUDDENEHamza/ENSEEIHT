class A {
	public int v;
	
	public void setV(int _v) {
		this.v = _v;
	}

	
	public int getV() {
		return this.v;
	}
	
	
	public A(int _v) {
		this.v = _v;
	}

	public A() {
		this.v = 10;
	}
	
}

public class Main {

	public static void main(String args[]) {
		A a2 = new A( 5 );
		A a1 = new A( 6 );
		print a2.getV();
		a1 = a2;
		print a1.getV();
		a1.setV(1);
	}
}
